const bs58 = require('bs58');
const contract = 'auth.iost';
function checkPublicKey(key) {
    let b = bs58.decode(key);
    return b.length === 32;
}
module.exports = (iost) => {
    return {
        /**
         * 
         * @param {string} id 
         * @param {string} permision 
         * @param {string} publicKey 
         * @param {number} threshold 
         * @param {Transaction.Tx} tx 
         * @returns {Transaction.Tx}
         */
        assignPermission(id, permision, publicKey, threshold, tx = iost.newEmptyTx()) {
            if (!checkPublicKey(publicKey)) {
                throw new Error('invalid public key')
            }
            iost.call(contract, 'assignPermission', [id, permision, publicKey, threshold], tx);
            return tx
        },
        /**
         * 
         * @param {string} id 
         * @param {string} permision 
         * @param {string} publicKey 
         * @param {Transaction.Tx} tx 
         * @returns {Transaction.Tx}
         */
        revokePermission(id, permision, publicKey, tx = iost.newEmptyTx()) {
            if (!checkPublicKey(publicKey)) {
                throw new Error('invalid public key')
            }
            iost.call(contract, 'revokePermission', [id, permision, publicKey], tx);
            return tx
        },
        /**
         * 
         * @param {string} name 
         * @param {string} ownerkey 
         * @param {string} activekey 
         * @param {Transaction.Tx} tx 
         * @returns {Transaction.Tx}
         */
        signUp(name, ownerkey, activekey, tx = iost.newEmptyTx()) {
            if (!checkPublicKey(ownerkey) || !checkPublicKey(activekey)) {
                throw new Error('invalid public key')
            }
            iost.call(contract, 'signUp', [name, ownerkey, activekey], tx);
            return tx
        }
    }
}
